        void printFibo(int aj)
        {
            static long int first = 0, second = 1, sum;
            if(aj > 1)
            {
                sum = first + second;
                first = second;
                second = sum;
                printf("%ld ", sum);
                printFibo(aj-1);    // recursive call
            }
            else
            {
                // after the elements, for line break
                printf("\n\n\n");
            }
        }
